WHERE {
{
?conference a bibo:Conference.
}UNION{

}
?role bfo:BFO_0000054 ?conference .
?role a ?roleClass ;
roh:roleOf ?researcher .
FILTER NOT EXISTS {
?role a ?otherClass .
?otherClass rdfs:subClassOf ?roleClass .
FILTER (?otherClass != ?roleClass)
}
}
