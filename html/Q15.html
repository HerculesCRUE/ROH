prefix roh : <http://w3id.org/roh#>.
prefix vivo : <http://w3id.org/roh/mirror/vivo#>.
prefix uneskos-individuals : <http://w3id.org/roh/unesco-individuals#>.
SELECT DISTINCT ?project
WHERE {

{
?project a vivo:Project ;
roh:hasKnowledgeArea ?knowledgeArea .
?knowledgeArea skos:broader+|skos:narrower+|skos:related+ uneskos-individuals:1203 .
} UNION {
uneskos-individuals:1203 skos:broader+ ?topKnowledgeArea .
?topKnowledgeArea skos:narrower+ ?otherKnowledgeArea .
?project roh:hasKnowledgeArea ?otherKnowledgeArea ;
a vivo:Project .
}
}
