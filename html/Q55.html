?project a vivo:Project ;
vivo:relates ?role ;
roh:isSupportedBy ?funding ;

roh:fundedBy ?fundingProgram ;
roh:publicFunding "false"^^xsd:boolean .
?fundingProgram roh:promotedBy ?company .

?projectStatus a ?projectStatusClass .

}
